// FIRESTORE SECURITY RULES — Afterglow Music Dashboard
// Copy paste ke: Firebase Console → Firestore Database → Rules
// ══════════════════════════════════════════════════════════════

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Helper: check user is admin
    function isAdmin() {
      return request.auth != null &&
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }

    // Helper: check authenticated
    function isAuth() {
      return request.auth != null;
    }

    // Helper: check field not empty string
    function isNotEmpty(field) {
      return request.resource.data[field] is string &&
             request.resource.data[field].size() > 0;
    }

    // ── USERS ──
    match /users/{userId} {
      allow read: if isAuth() && (request.auth.uid == userId || isAdmin());
      allow write: if isAuth() && (request.auth.uid == userId || isAdmin());
    }

    // ── SUBMISSIONS ──
    match /submissions/{submissionId} {
      allow read: if isAuth();
      allow create: if isAuth() &&
                      request.resource.data.userId == request.auth.uid &&
                      isNotEmpty('title') &&
                      isNotEmpty('artist') &&
                      isNotEmpty('userEmail') &&
                      isNotEmpty('genre') &&
                      isNotEmpty('format') &&
                      isNotEmpty('coverImage') &&
                      isNotEmpty('spotifyUrl') &&
                      isNotEmpty('appleMusicUrl') &&
                      isNotEmpty('releaseDate') &&
                      request.resource.data.status == 'pending' &&
                      request.resource.data.tracks is int &&
                      request.resource.data.tracks > 0;
      allow update: if isAuth() &&
                      (resource.data.userId == request.auth.uid || isAdmin());
      allow delete: if isAuth() &&
                      (resource.data.userId == request.auth.uid || isAdmin());
    }

    // ── NOTIFICATIONS ──
    // Users can read/update their own, admin can write all
    match /notifications/{notifId} {
      allow read: if isAuth() &&
                    (resource.data.userId == request.auth.uid || isAdmin());
      allow create: if isAuth();
      allow update: if isAuth() &&
                      (resource.data.userId == request.auth.uid || isAdmin());
      allow delete: if isAdmin();
    }

    // ── PRESAVES ──
    // Public create (fans don't need to login), auth read
    match /presaves/{presaveId} {
      allow read: if isAuth();
      allow create: if true;
    }

    // ── ROYALTIES ──
    // Users read their own, admin write all
    match /royalties/{royaltyId} {
      allow read: if isAuth() &&
                    (resource.data.userId == request.auth.uid || isAdmin());
      allow create: if isAdmin();
      allow update: if isAdmin();
      allow delete: if isAdmin();
    }

    // ── ROYALTY UPLOADS (history log) ──
    match /royaltyUploads/{uploadId} {
      allow read: if isAdmin();
      allow create: if isAdmin();
      allow update: if isAdmin();
      allow delete: if isAdmin();
    }

  }
}
